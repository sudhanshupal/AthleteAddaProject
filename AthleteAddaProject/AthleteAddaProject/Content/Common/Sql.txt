ALTER PROCEDURE [dbo].[GetNewfeeds] 
@SearchDateTime datetime,
@SearchText nvarchar(100),
@SortBy int,
@NewsfeedCountFrom int,
@GetTotalNewsfeeds int,
@IsDateIncluded bit
AS
BEGIN
	SET NOCOUNT ON;
	if (@NewsfeedCountFrom = 0) 
	begin
	if (@SearchText = '')
		begin
			if(@IsDateIncluded = 0)
				begin
					select Top(3)n.Content, n.NewsFeed_Id, n.Datetime, n.Publisher_Id, n.Image_Path, n.IsPublished, n.IsDeleted, n.IsVerified,
					n.Comment_by_Reviewer, p.Display_Name , n.Title
					from Newsfeed n, Publisher p
					where n.Publisher_Id = p.Publisher_Id	
					and n.IsDeleted = 0	
					and n.IsVerified = 1
					and n.IsVerified = 1
				end
				else
				begin
					select Top(3)n.Content, n.NewsFeed_Id, n.Datetime, n.Publisher_Id, n.Image_Path, n.IsPublished, n.IsDeleted, n.IsVerified,
					n.Comment_by_Reviewer, p.Display_Name , n.Title
					from Newsfeed n, Publisher p
					where n.Publisher_Id = p.Publisher_Id
					and n.IsDeleted = 0	
					and n.IsVerified = 1
					and n.IsVerified = 1
					and convert(date, n.Datetime) = convert(date,@SearchDateTime)
				end
		end
		else 
		begin
			if(@IsDateIncluded = 0)
			begin
				select Top(3)n.Content, n.NewsFeed_Id, n.Datetime, n.Publisher_Id, n.Image_Path, n.IsPublished, n.IsDeleted, n.IsVerified,
				n.Comment_by_Reviewer, p.Display_Name , n.Title
				from Newsfeed n, Publisher p
				where n.Publisher_Id = p.Publisher_Id
				and n.IsDeleted = 0	
				and n.IsVerified = 1
				and n.IsVerified = 1
			end
			else
			begin
				select Top(3)n.Content, n.NewsFeed_Id, n.Datetime, n.Publisher_Id, n.Image_Path, n.IsPublished, n.IsDeleted, n.IsVerified,
				n.Comment_by_Reviewer, p.Display_Name , n.Title
				from Newsfeed n, Publisher p
				where n.Publisher_Id = p.Publisher_Id
				and n.IsDeleted = 0	
				and n.IsVerified = 1
				and n.IsVerified = 1
				and convert(date, n.Datetime) = convert(date,@SearchDateTime)
				and n.Content like '%'+@SearchText+'%'
			end
		end
	end
	else
	begin
		declare @condition nvarchar(300)

		if (@SearchText = '')
		begin
			if(@IsDateIncluded = 0)
			begin
				select n.Content, n.NewsFeed_Id, n.Datetime, n.Publisher_Id, n.Image_Path, n.IsPublished, n.IsDeleted, n.IsVerified,
				n.Comment_by_Reviewer, p.Display_Name, n.Title
				from Newsfeed n, Publisher p
				where n.Publisher_Id = p.Publisher_Id
				and n.IsDeleted = 0	
				and n.IsVerified = 1
				and n.IsVerified = 1
				Order by n.Datetime desc
				OFFSET @NewsfeedCountFrom ROWS
				FETCH NEXT @GetTotalNewsfeeds ROWS ONLY		 
			end
			else
			begin
				select n.Content, n.NewsFeed_Id, n.Datetime, n.Publisher_Id, n.Image_Path, n.IsPublished, n.IsDeleted, n.IsVerified,
				n.Comment_by_Reviewer, p.Display_Name, n.Title
				from Newsfeed n, Publisher p
				where n.Publisher_Id = p.Publisher_Id
				and n.IsDeleted = 0	
				and n.IsVerified = 1
				and n.IsVerified = 1
				and convert(date, n.Datetime) = convert(date,@SearchDateTime)
				Order by n.Datetime desc
				OFFSET @NewsfeedCountFrom ROWS
				FETCH NEXT @GetTotalNewsfeeds ROWS ONLY
			end
		end
		else
		begin
			if(@IsDateIncluded = 0)
			begin
				select n.Content, n.NewsFeed_Id, n.Datetime, n.Publisher_Id, n.Image_Path, n.IsPublished, n.IsDeleted, n.IsVerified,
				n.Comment_by_Reviewer, p.Display_Name, n.Title
				from Newsfeed n, Publisher p
				where n.Publisher_Id = p.Publisher_Id
				and n.IsDeleted = 0	
				and n.IsVerified = 1
				and n.IsVerified = 1
				and n.Content like '%'+@SearchText+'%'
				Order by n.Datetime desc
				OFFSET @NewsfeedCountFrom ROWS
				FETCH NEXT @GetTotalNewsfeeds ROWS ONLY
			end
			else
			begin
				select n.Content, n.NewsFeed_Id, n.Datetime, n.Publisher_Id, n.Image_Path, n.IsPublished, n.IsDeleted, n.IsVerified,
				n.Comment_by_Reviewer, p.Display_Name, n.Title
				from Newsfeed n, Publisher p
				where n.Publisher_Id = p.Publisher_Id
				and n.IsDeleted = 0	
				and n.IsVerified = 1
				and n.IsVerified = 1
				and convert(date, n.Datetime) = convert(date,@SearchDateTime)
				and n.Content like '%'+@SearchText+'%'
				Order by n.Datetime desc
				OFFSET @NewsfeedCountFrom ROWS
				FETCH NEXT @GetTotalNewsfeeds ROWS ONLY
			end
		end
	end
END
